    $(function () {
        var form         = $("#appointment_form");
        var site         = $('[name="site_id"]');
        var document     = $('[name="document_id"]');
        var operation    = $('[name="operation_id"]');
        var district     = $('[name="district_id"]');
        var container    = $('.b-container');
        var useDistricts = true;

        site.on('change', function (ev) {
            if (!$(this).val())
                return;

            // ?????????? ???????
            if ($(this).val() == 29) {
                $('#step-1').hide();
                $('#karelia').removeClass('hidden');
            } else {
                $('#step-1').show();
                $('#karelia').addClass('hidden');
            }

            $.ajax({
                url: '/services/appointment/check_use_districts/',
                type: 'get',
                data: form.serialize(),
                dataType: 'json',
                success: function (data) {
                    useDistricts = data;

                    if(!useDistricts) {
                        district.val(null);
                        district.closest('.cf-item').hide();
                        getDocuments();
                        return;
                    }

                    district.closest('.cf-item').show();
                    getDistricts();
                }
            });
        });

        district.on('change', function (ev) {
            if (!$(this).val())
                return;

            getDocuments();
        });

        document.on('change', function (ev) {
            if (!$(this).val())
                return;

            getOperations();
        });

        operation.on('change', function (ev) {
            if (!$(this).val())
                return;

            getSchedule();
        });

        form.on('submit', function (ev) {
            ev.preventDefault();

            form.addClass('t-form_disabled');
            $('#submit').addClass('b-loading t-small_cover_green');

            clear_errors();

            $.ajax({
                url: '/services/appointment/',
                type: 'post',
                data: form.serialize(),
                dataType: 'json',
                success: function (data) {
                    $('.b-captcha-image').children('img').attr('src', '/captcha' + '?' + Math.random());
                    $('[name="captcha"]').val('');
                    $('#submit').removeClass('b-loading t-small_cover_green');
                    form.removeClass('t-form_disabled');

                    if (data['errors']) {
                        show_errors(data['errors']);
                        return;
                    }

                    container.html(data);
                }
            });
        });

        $('body').on('focus', '.select-container, .input', function () {
            $(this).parent().find('.form-error').fadeOut(500, function () {
                $(this).remove();
            });
        });

        function getDistricts() {
            $.ajax({
                url: '/services/appointment/appointment_districts/',
                type: 'get',
                data: form.serialize(),
                dataType: 'json',
                success: function (data) {
                    district.find('option:gt(0)').remove();
                    if (data) {
                        for (var key in data) {
                            district.append($('<option/>', {value: key, text: data[key]}));
                        }
                    }
                    district.select('destroy').select({title: true});
                    district.prop('disabled', false);
                },
            });
        }

        function getDocuments() {
            $('#step-2').toggleClass('hidden', false);
            $('#pair_conditional').toggleClass('hidden', false);
            $.ajax({
                url: '/services/appointment/appointment_documents/',
                type: 'get',
                data: form.serialize(),
                dataType: 'json',
                success: function (data) {
                    document.find('option:gt(0)').remove();
                    for (var key in data) {
                        document.append($('<option/>', {value: key, text: data[key]}));
                    }
                    document.select('destroy').select({title: true});
                },
            });
        }

        function getOperations() {
            $.ajax({
                url: '/services/appointment/appointment_operations/',
                type: 'get',
                data: form.serialize(),
                dataType: 'json',
                success: function (data) {
                    operation.find('option:gt(0)').remove();
                    for (var key in data) {
                        operation.append($('<option/>', {value: key, text: data[key]}));
                    }
                    operation.select('destroy').select({title: true});
                },
            });
        }

        function getSchedule() {
            $("#step-4").html('');
            $.ajax({
                url: '/services/appointment/appointment_schedule_view/',
                type: 'get',
                data: form.serialize(),
                dataType: 'html',
                success: function (data) {
                    $("#step-4").html($(data));
                },
            });
        }

        function clear_errors() {
            $('.form-error').remove();
        }

        function show_errors(errors) {
            var parent   = {};
            var template = '';

            for (var key in errors) {
                switch(key) {
                    case 'date':
                    case 'time':
                    case 'ip':
                    case 'service_handler':
                        parent = $('.appointment-date');
                        break;
                    default:
                        parent = $('[name=' + key + ']');
                }

                parent.closest('.cf-item').before('<p class="form-error">' + errors[key] + '</p>');
            }
        }
    });
